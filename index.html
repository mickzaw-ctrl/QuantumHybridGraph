Quickstart

Up and running in 60 seconds

git clone https://github.com/mickzaw-ctrl/QuantumHybridGraph.git
cd QuantumHybridGraph
pip install -e .

# Basic usage
python -c "
from quantum_hybrid_graph import QuantumHybridGraph
kernel = QuantumHybridGraph(initial_nodes=10, clique_threshold=3, seed=42)
for _ in range(10):
    kernel.run_cycle()
print(kernel.summary())
"

# QAOA MaxClique on local simulator
pip install -e .[quantum]
python -c "
from quantum_hybrid_graph import QuantumHybridGraph
kernel = QuantumHybridGraph(initial_nodes=12, seed=0)
for _ in range(15): kernel.run_cycle()
result = kernel.qaoa_maxclique(reps=2, optimizer='COBYLA', shots=1024)
print(result)
"
